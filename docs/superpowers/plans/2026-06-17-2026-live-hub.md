```ts
// KST 표시 라벨. Intl로 Asia/Seoul 변환 (서버/클라 동일 결과).
export function toKstLabel(kickoffUtc: string): string {
  const d = new Date(kickoffUtc)
  const parts = new Intl.DateTimeFormat('ko-KR', {
    timeZone: 'Asia/Seoul', month: 'numeric', day: 'numeric',
    weekday: 'short', hour: '2-digit', minute: '2-digit', hour12: false,
  }).formatToParts(d)
  const g = (t: string) => parts.find((p) => p.type === t)?.value ?? ''
  return `${g('month')}/${g('day')} ${g('weekday')} ${g('hour')}:${g('minute')}`
}

// KST 날짜 키 (YYYY-MM-DD) — 날짜별 그룹핑용.
export function kstDateKey(kickoffUtc: string): string {
  const parts = new Intl.DateTimeFormat('en-CA', {
    timeZone: 'Asia/Seoul', year: 'numeric', month: '2-digit', day: '2-digit',
  }).format(new Date(kickoffUtc))
  return parts // en-CA → "2026-06-19"
}
```
(주의: `formatToParts`의 weekday가 "금" 또는 "금요일"일 수 있음 — `weekday:'short'`는 "금". 실패 시 테스트 기대값을 실제 출력에 맞춰 조정하되, 형식 일관성 유지.)
